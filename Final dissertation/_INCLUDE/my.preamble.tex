\newcommand{\pageHeaderNumPos}[1]{
    \fancypagestyle{plain}{
    \fancyhf{}
    % Hide line
    \renewcommand{\headrulewidth}{0pt}
    \renewcommand{\footrulewidth}{0pt}
    % \fancyhead[#1]{\thepage}

    \fancyhead[OR]{\thepage}
