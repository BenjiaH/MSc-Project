% Set the position of the page number
\usepackage{fancyhdr}
\newcommand{\pageHeaderNumPos}[1]{
    \fancypagestyle{plain}{
    \fancyhf{}
    % Hide line
    \renewcommand{\headrulewidth}{0pt}
    \renewcommand{\footrulewidth}{0pt}
    \fancyhead[#1]{\thepage}}
    \pagestyle{plain}}

\newcommand{\pageFooterNumPos}[1]{
    \fancypagestyle{plain}{
    \fancyhf{}
    % Hide line
    \renewcommand{\headrulewidth}{0pt}
    \renewcommand{\footrulewidth}{0pt}
    \fancyfoot[#1]{\thepage}}
    \pagestyle{plain}}
